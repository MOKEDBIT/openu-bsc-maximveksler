public static int descendant (BinaryNode root, int x , int y)

{

  if (root == null)

     return 0;

  if (root._data == x)

     return ((isInTree (root._left , y)) ||

             (isInTree (root._right , y)));

  return (descendant (root._left, x, y)) ||

         (descendant (root._right, x, y));

}

public static int min (int a, int b, int c)

// The function gets 3 numbers and returns the minimum.

{

  if ((a < b) && (a < c))

       return a;

  else if ((b < a) && (b < c)

          return b;

       else

          return c;

}

public static int minNode (BinaryNode root)

{

  int num;

 

  num = root._data;

  if (isLeaf (root))

     return num;

  if (root._right == null)

     return (min (num, minNode (root._left)));

  if (root._left == null)

     return (min (num, minNode (root._right)));

  return (min (num ,

               minNode (root._left),

               minNode (root._right)));

}

{

  if (root == null)

       return -1;
