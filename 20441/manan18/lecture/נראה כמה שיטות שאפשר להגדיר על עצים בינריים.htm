·               השיטה הבאה מקבלת שורש של עץ בינרי וסופרת כמה צמתים (קדקודים) יש בעץ.

public static int numOfNodes (BinaryNode root)

// מונה את מספר הצמתים בעץ.

{

  if (root == null)

    return 0;

  return (1 + numOfNodes(root._left) +

              numOfNodes(root._right));

}

public static int numOfLeaves (BinaryNode root)

// מונה את מספר העלים בעץ

{

    return 0;

    return 1;

·               השיטה הבאה מקבלת צומת (קדקוד) ובודקת אם הוא עלה.

public static int isLeaf (BinaryNode current)

// מקבל צומת ומחזיר תשובה אם הוא עלה

{

  return ((current._left == null) && (current.right == null));

}

 

·               בהנחה שטיפוס הנתונים שמאוחסן בעץ הבינרי הוא מספרים שלמים, השיטה הבאה מקבלת שורש של עץ בינרי ומספר שלם, ובודקת האם המספר נמצא בעץ. כלומר, ProblemData הוא int

public static int isInTree (BinaryNode current, int num)

// מקבל מספר ועץ (לא ממוין) ומחזיר תשובה אם המספר נמצא בעץ

{

  if (current == null)

     return 0;

 

  if (current._data == num)

     return 1;

 

  return (isInTree (current._left, num)) ||

         (isInTree (current._right, num));

}
